# Prompt — Simulação E2E Clínica Multi-Especialidade (v3)
## Foco: Empatia, Protocolo de Atendimento e PIX por Profissional

## Objetivo real desta simulação
Verificar se a IA:
1. Tem EMPATIA e tom adequado no início de cada conversa
2. Aplica PROTOCOLO DIFERENTE para lead particular x convênio x paciente da clínica
3. Envia a CHAVE PIX correta do dentista certo após agendamento (quando pixMode=required)
4. COBRA ou NÃO a consulta conforme configuração individual de cada profissional
5. Respeita as configurações INDIVIDUAIS de cada dentista (não usa configuração genérica)

### BLOCO 1 — EMPATIA E PRIMEIRO CONTATO
### ══════════════════════════════════════════════

**Cenário 1 — Empatia: lead com dor**
```
Lead: "oi, to com muita dor de dente"
Esperado:
  ✓ IA demonstra empatia ANTES de qualquer oferta de agendamento
  ✓ Pergunta sobre a dor / acolhe a situação
  ✓ Menciona urgência / possibilidade de encaixe
  ✗ NÃO deve começar já perguntando convênio ou particular
  ✗ NÃO deve começar já oferecendo horários

Verificar na resposta:
  - Contém palavras de acolhimento: "entendo", "sinto", "vamos te ajudar",
    "não se preocupe", "cuidar", "atender" ou similar
  - NÃO começa com "Você usa convênio ou paga particular?"
```

**Cenário 2 — Empatia: lead com medo de dentista**
```
Lead: "preciso ir ao dentista mas tenho muito medo"
Esperado:
  ✓ IA valida o medo sem minimizar ("entendo, muitas pessoas sentem isso")
  ✓ Menciona que a equipe é preparada para atender com cuidado
  ✓ Só depois oferece agendamento
  ✗ NÃO ignora o medo e vai direto para horários

Verificar:
  - Resposta contém acolhimento do medo
  - Não vai direto para triagem de convênio
```

**Cenário 3 — Primeiro contato neutro, triagem correta**
```
Lead: "oi, quero marcar uma consulta"
Esperado:
  ✓ IA se apresenta como Sofia da Clínica OdontoVida
```

## Helpers de verificação — criar no arquivo de teste

```typescript
// Verifica empatia na resposta
function assertHasEmpathy(response: string, context: string): void {
  const empathyWords = [
    "entendo", "compreendo", "sinto", "imagino", "sei como",
    "não se preocupe", "vamos te ajudar", "pode contar",
    "estamos aqui", "com cuidado", "nossa equipe",
    "que bom", "fico feliz", "obrigado por"
  ];
  const hasEmpathy = empathyWords.some(w =>
    response.toLowerCase().includes(w)
  );
  assert(hasEmpathy,
    `[${context}] IA não demonstrou empatia. Resposta: "${response.substring(0, 150)}"`
  );
}

// Verifica que NÃO fez triagem de convênio para paciente conhecido
function assertNoInsuranceTriage(response: string, context: string): void {
  const triageWords = [
    "convênio ou particular", "plano ou particular",
    "usa convênio", "tem plano", "como vai pagar"
  ];
  const hasTriage = triageWords.some(w =>
    response.toLowerCase().includes(w)
  );
  assert(!hasTriage,
    `[${context}] IA fez triagem de convênio para paciente conhecido. Resposta: "${response.substring(0, 150)}"`
  );
}

// Verifica chave PIX correta por dentista
function assertCorrectPixKey(
  response: string,
  professional: "ana" | "roberto" | "marcos",
  context: string
): void {
  const pixKeys = {
    ana: "ana@odontovida.com.br",
    roberto: "11999887766",
    marcos: null, // não tem PIX
  };
  const expectedKey = pixKeys[professional];

  if (professional === "marcos") {
    // Dr. Marcos não tem PIX — não deve aparecer chave nenhuma
    assert(
      !response.includes("pix") && !response.includes("PIX"),
      `[${context}] Dr. Marcos não tem PIX mas IA enviou chave. Resposta: "${response.substring(0, 150)}"`
    );
    return;
  }

  if (expectedKey) {
    assert(
      response.includes(expectedKey),
      `[${context}] Chave PIX incorreta ou ausente. Esperado: "${expectedKey}". Resposta: "${response.substring(0, 150)}"`
    );
  }
}

// Verifica cobrança de consulta por dentista
function assertConsultationFee(
  response: string,
  professional: "ana" | "roberto" | "marcos_particular",
  context: string
): void {
  const fees = {
    ana: "150",
    roberto: null,  // gratuita
    marcos_particular: "200",
  };
  const fee = fees[professional];

  if (!fee) {
    // Consulta gratuita — verificar que menciona gratuidade
    const gratuitaWords = ["gratuita", "grátis", "sem custo", "sem cobrança", "cortesia"];
    const hasGratuita = gratuitaWords.some(w => response.toLowerCase().includes(w));
    assert(hasGratuita,
      `[${context}] Dr. Roberto tem consulta gratuita mas IA não mencionou. Resposta: "${response.substring(0, 150)}"`
    );
  } else {
    assert(
      response.includes(fee),
      `[${context}] Valor da consulta R$${fee} não mencionado. Resposta: "${response.substring(0, 150)}"`
    );
  }
}
```
