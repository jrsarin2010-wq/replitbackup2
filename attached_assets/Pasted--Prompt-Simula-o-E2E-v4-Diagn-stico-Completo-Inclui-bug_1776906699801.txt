## Helpers de verificação

```typescript
// Verifica empatia antes de triagem
function assertEmpathyBeforeTriage(response: string, context: string): void {
  const empathyWords = [
    "entendo", "imagino", "que situação", "que chato",
    "sinto muito", "poxa", "caramba", "nossa",
    "não se preocupe", "vamos cuidar", "fica tranquilo"
  ];
  const triageWords = [
    "plano ou particular", "convênio ou particular",
    "vai usar plano", "tem plano"
  ];
  const hasEmpathy = empathyWords.some(w => response.toLowerCase().includes(w));
  const hasTriageInFirst = triageWords.some(w => response.toLowerCase().includes(w));
  
  // Empatia é obrigatória quando contexto tem dor/medo
  // Triagem na PRIMEIRA resposta é proibida quando há dor/medo
  if (!hasEmpathy) {
    console.warn(`[${context}] AVISO: Sem empatia detectada`);
  }
  assert(!hasTriageInFirst,
    `[${context}] FALHA: Perguntou plano/particular na primeira resposta sem acolher`
  );
}

// Verifica que PIX correto foi enviado
function assertPixKey(response: string, expectedKey: string, context: string): void {
  assert(
    response.includes(expectedKey),
    `[${context}] FALHA: Chave PIX incorreta. Esperado: "${expectedKey}"`
  );
}

// Verifica ausência de termos de venda em modo convênio
function assertNoSalesTerms(response: string, context: string): void {
  const forbidden = [
    "última vaga", "ultima vaga", "agenda disputada",
    "corre que", "garante agora", "não perca", "nao perca",
    "consegui um encaixe", "encaixe especial", "aproveite",
    "oportunidade", "vaga disputada"
  ];
  for (const term of forbidden) {
    assert(
      !response.toLowerCase().includes(term),
      `[${context}] FALHA: Termo de venda proibido em modo convênio: "${term}"`
    );
  }
}

// Verifica que não há cobrança para convênio
function assertNoChargeForInsurance(response: string, context: string): void {
  const chargeTerms = ["R$150", "R$200", "R$300", "pix", "PIX", "pagar antes"];
  for (const term of chargeTerms) {
    assert(
      !response.toLowerCase().includes(term.toLowerCase()),
      `[${context}] FALHA: IA cobrou de paciente convênio: "${term}"`
    );
  }
}
```

## Instruções para a Replit

1. Seguir EXATAMENTE o padrão do `e2e-test.ts` existente

2. Setup: criar tenant + 3 profissionais com TODOS os campos listados acima

3. Para cenários de paciente (Bloco 5), criar paciente no banco antes

4. **MAIS IMPORTANTE**: mostrar a RESPOSTA COMPLETA DA IA em cada cenário
   nos logs e no relatório final — não apenas pass/fail

5. Para cenários de modo (Bloco 1), capturar o `mode_resolved` dos logs
   do servidor durante a execução — esse é o dado mais crítico

6. Teardown apaga TUDO do tenant de teste

7. Rodar testes unitários ANTES da simulação:
   `pnpm --filter @workspace/api-server run test`

8. Após criar o arquivo, rodar IMEDIATAMENTE e reportar o diagnóstico
   com todas as respostas reais
